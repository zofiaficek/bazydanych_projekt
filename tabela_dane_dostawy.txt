CREATE TABLE dane_dostawy (
    id_dostawy INTEGER NOT NULL,
    id_produktu INTEGER NOT NULL,
    ilosc INTEGER NOT NULL CHECK (ilosc > 0),
    cena_zakupu DECIMAL(10,2) NOT NULL CHECK (cena_zakupu > 0),
    
    PRIMARY KEY (id_dostawy, id_produktu),
	
    CONSTRAINT fk_dostawy
        FOREIGN KEY (id_dostawy) 
        REFERENCES dostawy(id_dostawy) 
        ON DELETE CASCADE,

    CONSTRAINT fk_produkty
        FOREIGN KEY (id_produktu) 
        REFERENCES produkty(id_produktu)
        ON DELETE CASCADE
);
