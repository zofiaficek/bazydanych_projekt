CREATE TABLE dane_zamowienia (
    zamowienie_id INTEGER NOT NULL,
    produkt_id INTEGER NOT NULL,
    ilosc INTEGER NOT NULL CHECK (ilosc > 0),
    cena_zakupu DECIMAL(10,2) NOT NULL CHECK (cena_zakupu > 0),
    
    CONSTRAINT fk_zamowienia
        FOREIGN KEY (zamowienie_id) 
        REFERENCES zamowienia(zamowienie_id)
        ON DELETE CASCADE,
        
    CONSTRAINT fk_produkty
        FOREIGN KEY (produkt_id) 
        REFERENCES produkty(produkt_id)
        ON DELETE CASCADE
);
